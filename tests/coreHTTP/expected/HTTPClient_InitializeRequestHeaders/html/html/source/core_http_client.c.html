550 * and n must be less than the actual size of either string.
551 *
552 * @param[in] str1 First string to be compared.
553 * @param[in] str2 Second string to be compared.
554 * @param[in] n The maximum number of characters to be compared.
555 *
556 * @return One of the following:
557 * 0 if str1 is equal to str2
558 * 1 if str1 is not equal to str2.

562 size_t n );

573 static int8_t caseInsensitiveStringCmp( const char * str1,
574 const char * str2,
575 size_t n )
576 {
577 size_t i = 0U;
578
579 for( i = 0U; i < n; i++ )
580 {
581 if( toupper( str1[ i ] ) != toupper( str2[ i ] ) )
582 {
583 break;
584 }
585 }
586
587 return ( i == n ) ? 0 : 1;
588 }

592 static void processCompleteHeader( HTTPParsingContext_t * pParsingContext )
593 {
594 HTTPResponse_t * pResponse = NULL;
595
596 assert( pParsingContext != NULL );
597 assert( pParsingContext->pResponse != NULL );
598
599 pResponse = pParsingContext->pResponse;
600
601 /* A header is complete when both the last header field and value have been
602 * filled in. */
603 if( ( pParsingContext->pLastHeaderField != NULL ) &&
604 ( pParsingContext->pLastHeaderValue != NULL ) )
605 {
606 assert( pResponse->headerCount < SIZE_MAX );
607 /* Increase the header count. */
608 pResponse->headerCount++;
609
610 LogDebug( ( "Response parsing: Found complete header: "
611 "HeaderField=%.*s, HeaderValue=%.*s",
612 ( int ) ( pParsingContext->lastHeaderFieldLen ),
613 pParsingContext->pLastHeaderField,
614 ( int ) ( pParsingContext->lastHeaderValueLen ),
615 pParsingContext->pLastHeaderValue ) );
616
617 /* If the application registered a callback, then it must be notified. */
618 if( pResponse->pHeaderParsingCallback != NULL )
619 {
620 pResponse->pHeaderParsingCallback->onHeaderCallback(
621 pResponse->pHeaderParsingCallback->pContext,
622 pParsingContext->pLastHeaderField,
623 pParsingContext->lastHeaderFieldLen,
624 pParsingContext->pLastHeaderValue,
625 pParsingContext->lastHeaderValueLen,
626 pResponse->statusCode );
627 }
628
629 /* Prepare the next header field and value for the first invocation of
630 * httpParserOnHeaderFieldCallback() and
631 * httpParserOnHeaderValueCallback(). */
632 pParsingContext->pLastHeaderField = NULL;
633 pParsingContext->lastHeaderFieldLen = 0U;
634 pParsingContext->pLastHeaderValue = NULL;
635 pParsingContext->lastHeaderValueLen = 0U;
636 }
637 }

674 assert( pResponse != NULL );
675
676 /* Set the location of what to parse next. */
677 pParsingContext->pBufferCur = pLoc + length;
678
679 /* Initialize the first header field and value to be passed to the user
680 * callback. */
681 pParsingContext->pLastHeaderField = NULL;
682 pParsingContext->lastHeaderFieldLen = 0U;
683 pParsingContext->pLastHeaderValue = NULL;
684 pParsingContext->lastHeaderValueLen = 0U;
685
686 /* httpParserOnStatusCallback() is reached because http_parser_execute() has
687 * successfully read the HTTP response status code. */
688 pResponse->statusCode = ( uint16_t ) ( pHttpParser->status_code );
689
690 LogDebug( ( "Response parsing: Found the Reason-Phrase: "
691 "StatusCode=%u, ReasonPhrase=%.*s",
692 ( unsigned int ) pResponse->statusCode,
693 ( int ) length,
694 pLoc ) );
695
696 return HTTP_PARSER_CONTINUE_PARSING;
697 }

701 static int httpParserOnHeaderFieldCallback( http_parser * pHttpParser,
702 const char * pLoc,
703 size_t length )
704 {
705 HTTPParsingContext_t * pParsingContext = NULL;
706 HTTPResponse_t * pResponse = NULL;
707
708 assert( pHttpParser != NULL );
709 assert( pHttpParser->data != NULL );
710 assert( pLoc != NULL );
711
712 pParsingContext = ( HTTPParsingContext_t * ) ( pHttpParser->data );
713 pResponse = pParsingContext->pResponse;
714
715 assert( pResponse != NULL );
716
717 /* If this is the first time httpParserOnHeaderFieldCallback() has been
718 * invoked on a response, then the start of the response headers is NULL. */
719 if( pResponse->pHeaders == NULL )
720 {
721 pResponse->pHeaders = ( const uint8_t * ) pLoc;
722 }
723
724 /* Set the location of what to parse next. */
725 pParsingContext->pBufferCur = pLoc + length;
726
727 /* The httpParserOnHeaderFieldCallback() always follows the
728 * httpParserOnHeaderValueCallback() if there is another header field. When
729 * httpParserOnHeaderValueCallback() is not called in succession, then a
730 * complete header has been found. */
731 processCompleteHeader( pParsingContext );
732
733 /* If httpParserOnHeaderFieldCallback() is invoked in succession, then the
734 * last time http_parser_execute() was called only part of the header field
735 * was parsed. The indication of successive invocations is a non-NULL
736 * pParsingContext->pLastHeaderField. */
737 if( pParsingContext->pLastHeaderField == NULL )
738 {
739 pParsingContext->pLastHeaderField = pLoc;
740 pParsingContext->lastHeaderFieldLen = length;
741 }
742 else
743 {
744 assert( pParsingContext->lastHeaderFieldLen <= SIZE_MAX - length );
745 pParsingContext->lastHeaderFieldLen += length;
746 }
747
748 LogDebug( ( "Response parsing: Found a header field: "
749 "HeaderField=%.*s",
750 ( int ) length,
751 pLoc ) );
752
753 return HTTP_PARSER_CONTINUE_PARSING;
754 }

803 static int httpParserOnHeadersCompleteCallback( http_parser * pHttpParser )
804 {
805 int shouldContinueParse = HTTP_PARSER_CONTINUE_PARSING;
806 HTTPParsingContext_t * pParsingContext = NULL;
807 HTTPResponse_t * pResponse = NULL;
808
809 assert( pHttpParser != NULL );
810 assert( pHttpParser->data != NULL );
811
812 pParsingContext = ( HTTPParsingContext_t * ) ( pHttpParser->data );
813 pResponse = pParsingContext->pResponse;
814
815 assert( pResponse != NULL );
816 assert( pParsingContext->pBufferCur != NULL );
817
818 /* The current location to parse was updated in previous callbacks and MUST
819 * always be within the response buffer. */
820 assert( pParsingContext->pBufferCur >= ( const char * ) ( pResponse->pBuffer ) );
821 assert( pParsingContext->pBufferCur < ( const char * ) ( pResponse->pBuffer + pResponse->bufferLen ) );
822
823 /* `\r\n\r\n`, `\r\n\n`, `\n\r\n`, and `\n\n` are all valid indicators of
824 * the end of the response headers. To reduce complexity these characters
825 * are not included in the response headers length returned to the user. */
826
827 /* If headers existed, then pResponse->pHeaders was set during the first
828 * call to httpParserOnHeaderFieldCallback(). */
829 if( pResponse->pHeaders != NULL )
830 {
831 /* The start of the headers ALWAYS come before the the end of the headers. */
832 assert( ( const char * ) ( pResponse->pHeaders ) < pParsingContext->pBufferCur );
833
834 /* MISRA Rule 10.8 flags the following line for casting from a signed
835 * pointer difference to a size_t. This rule is suppressed because in
836 * in the previous statement it is asserted that the pointer difference
837 * will never be negative. */
838 /* coverity[misra_c_2012_rule_10_8_violation] */
839 pResponse->headersLen = ( size_t ) ( pParsingContext->pBufferCur - ( const char * ) ( pResponse->pHeaders ) );
840 }
841 else
842 {
843 pResponse->headersLen = 0U;
844 }
845
846 /* If the Content-Length header was found, then pHttpParser->content_length
847 * will not be equal to the maximum 64 bit integer. */
848 if( pHttpParser->content_length != UINT64_MAX )
849 {
850 pResponse->contentLength = ( size_t ) ( pHttpParser->content_length );
851 }
852 else
853 {
854 pResponse->contentLength = 0U;
855 }
856
857 /* If the Connection: close header was found this flag will be set. */
858 if( ( pHttpParser->flags & ( unsigned int ) ( F_CONNECTION_CLOSE ) ) != 0U )
859 {
860 pResponse->respFlags |= HTTP_RESPONSE_CONNECTION_CLOSE_FLAG;
861 }
862
863 /* If the Connection: keep-alive header was found this flag will be set. */
864 if( ( pHttpParser->flags & ( unsigned int ) ( F_CONNECTION_KEEP_ALIVE ) ) != 0U )
865 {
866 pResponse->respFlags |= HTTP_RESPONSE_CONNECTION_KEEP_ALIVE_FLAG;
867 }
868
869 /* http_parser_execute() requires that callback implementations must
870 * indicate that parsing stops on headers complete, if response is to a HEAD
871 * request. A HEAD response will contain Content-Length, but no body. If
872 * the parser is not stopped here, then it will try to keep parsing past the
873 * end of the headers up to the Content-Length found. */
874 if( pParsingContext->isHeadResponse == 1U )
875 {
876 shouldContinueParse = HTTP_PARSER_STOP_PARSING;
877 }
878
879 /* If headers are present in the response, then
880 * httpParserOnHeadersCompleteCallback() always follows
881 * the httpParserOnHeaderValueCallback(). When
882 * httpParserOnHeaderValueCallback() is not called in succession, then a
883 * complete header has been found. */
884 processCompleteHeader( pParsingContext );
885
886 LogDebug( ( "Response parsing: Found the end of the headers." ) );
887
888 return shouldContinueParse;
889 }

893 static int httpParserOnBodyCallback( http_parser * pHttpParser,
894 const char * pLoc,
895 size_t length )
896 {
897 int shouldContinueParse = HTTP_PARSER_CONTINUE_PARSING;
898 HTTPParsingContext_t * pParsingContext = NULL;
899 HTTPResponse_t * pResponse = NULL;
900 char * pNextWriteLoc = NULL;
901
902 assert( pHttpParser != NULL );
903 assert( pHttpParser->data != NULL );
904 assert( pLoc != NULL );
905
906 pParsingContext = ( HTTPParsingContext_t * ) ( pHttpParser->data );
907 pResponse = pParsingContext->pResponse;
908
909 assert( pResponse != NULL );
910 assert( pResponse->pBuffer != NULL );
911 assert( pLoc >= ( const char * ) ( pResponse->pBuffer ) );
912 assert( pLoc < ( const char * ) ( pResponse->pBuffer + pResponse->bufferLen ) );
913
914 /* If this is the first time httpParserOnBodyCallback() has been invoked,
915 * then the start of the response body is NULL. */
916 if( pResponse->pBody == NULL )
917 {
918 /* Ideally the start of the body should follow right after the header
919 * end indicating characters, but to reduce complexity and ensure users
920 * are given the correct start of the body, we set the start of the body
921 * to what the parser tells us is the start. This could come after the
922 * initial transfer encoding chunked header. */
923 pResponse->pBody = ( const uint8_t * ) ( pLoc );
924 pResponse->bodyLen = 0U;
925 }
926
927 /* The next location to write. */
928
929 /* MISRA Rule 11.8 flags casting away the const qualifier in the pointer
930 * type. This rule is suppressed because when the body is of transfer
931 * encoding chunked, the body must be copied over the chunk headers that
932 * precede it. This is done to have a contiguous response body. This does
933 * affect future parsing as the changed segment will always be before the
934 * next place to parse. */
935 /* coverity[misra_c_2012_rule_11_8_violation] */
936 pNextWriteLoc = ( char * ) ( pResponse->pBody + pResponse->bodyLen );
937
938 /* If the response is of type Transfer-Encoding: chunked, then actual body
939 * will follow the the chunked header. This body data is in a later location
940 * and must be moved up in the buffer. When pLoc is greater than the current
941 * end of the body, that signals the parser found a chunk header. */
942
943 /* MISRA Rule 18.3 flags pLoc and pNextWriteLoc as pointing to two different
944 * objects. This rule is suppressed because both pNextWriteLoc and pLoc
945 * point to a location in the response buffer. */
946 /* coverity[pointer_parameter] */
947 /* coverity[misra_c_2012_rule_18_3_violation] */
948 if( pLoc > pNextWriteLoc )
949 {
950 /* memmove is used instead of memcpy because memcpy has undefined behavior
951 * when source and destination locations in memory overlap. */
952 ( void ) memmove( pNextWriteLoc, pLoc, length );
953 }
954
955 /* Increase the length of the body found. */
956 pResponse->bodyLen += length;
957
958 /* Set the next location of parsing. */
959 pParsingContext->pBufferCur = pLoc + length;
960
961 LogDebug( ( "Response parsing: Found the response body: "
962 "BodyLength=%lu",
963 ( unsigned long ) length ) );
964
965 return shouldContinueParse;
966 }

1022 static HTTPStatus_t processHttpParserError( const http_parser * pHttpParser )
1023 {
1024 HTTPStatus_t returnStatus = HTTPSuccess;
1025
1026 assert( pHttpParser != NULL );
1027
1028 switch( ( enum http_errno ) ( pHttpParser->http_errno ) )
1029 {
1030 case HPE_OK:
1031 /* There were no errors. */
1032 break;
1033
1034 case HPE_INVALID_EOF_STATE:
1035
1036 /* In this case the parser was passed a length of zero, which indicates
1037 * an EOF from the server, in the middle of parsing the response.
1038 * This case is already handled by checking HTTPParsingContext_t.state. */
1039 break;
1040
1041 case HPE_HEADER_OVERFLOW:
1042 LogError( ( "Response parsing error: Header byte limit "
1043 "exceeded: HeaderByteLimit=%u",
1044 HTTP_MAX_RESPONSE_HEADERS_SIZE_BYTES ) );
1045 returnStatus = HTTPSecurityAlertResponseHeadersSizeLimitExceeded;
1046 break;
1047
1048 case HPE_CLOSED_CONNECTION:
1049 LogError( ( "Response parsing error: Data received past complete "
1050 "response with \"Connection: close\" header present." ) );
1051 returnStatus = HTTPSecurityAlertExtraneousResponseData;
1052 break;
1053
1054 case HPE_INVALID_CHUNK_SIZE:
1055
1056 /* http_parser_execute() does not give feedback on the exact failing
1057 * character and location. */
1058 LogError( ( "Response parsing error: Invalid character found in "
1059 "chunk header." ) );
1060 returnStatus = HTTPSecurityAlertInvalidChunkHeader;
1061 break;
1062
1063 case HPE_INVALID_VERSION:
1064
1065 /* http_parser_execute() does not give feedback on the exact failing
1066 * character and location. */
1067 LogError( ( "Response parsing error: Invalid character found in "
1068 "HTTP protocol version." ) );
1069 returnStatus = HTTPSecurityAlertInvalidProtocolVersion;
1070 break;
1071
1072 case HPE_INVALID_STATUS:
1073
1074 /* There could be an invalid character or the status code number
1075 * could be out of range. This feedback is not given back by the
1076 * http-parser library. */
1077 LogError( ( "Response parsing error: Invalid Status code." ) );
1078 returnStatus = HTTPSecurityAlertInvalidStatusCode;
1079 break;
1080
1081 case HPE_STRICT:
1082 case HPE_INVALID_CONSTANT:
1083 LogError( ( "Response parsing error: Invalid character found in "
1084 "Status-Line or header delimiters." ) );
1085 returnStatus = HTTPSecurityAlertInvalidCharacter;
1086 break;
1087
1088 case HPE_LF_EXPECTED:
1089 LogError( ( "Response parsing error: Expected line-feed in header "
1090 "not found." ) );
1091 returnStatus = HTTPSecurityAlertInvalidCharacter;
1092 break;
1093
1094 case HPE_INVALID_HEADER_TOKEN:
1095
1096 /* http_parser_execute() does not give feedback on the exact failing
1097 * character and location. */
1098 LogError( ( "Response parsing error: Invalid character found in "
1099 "headers." ) );
1100 returnStatus = HTTPSecurityAlertInvalidCharacter;
1101 break;
1102
1103 case HPE_INVALID_CONTENT_LENGTH:
1104
1105 /* http_parser_execute() does not give feedback on the exact failing
1106 * character and location. */
1107 LogError( ( "Response parsing error: Invalid character found in "
1108 "content-length headers." ) );
1109 returnStatus = HTTPSecurityAlertInvalidContentLength;
1110 break;
1111
1112 case HPE_UNEXPECTED_CONTENT_LENGTH:
1113 LogError( ( "Response parsing error: A Content-Length header was "
1114 "found when it shouldn't have been." ) );
1115 returnStatus = HTTPSecurityAlertInvalidContentLength;
1116 break;
1117
1118 /* All other error cases cannot be triggered and indicate an error in the
1119 * third-party parsing library if found. */
1120 default:
1121 LogError( ( "Error in third-party http-parser library." ) );
1122 returnStatus = HTTPParserInternalError;
1123 break;
1124 }
1125
1126 /* Errors with CB_ prepending are manual returns of non-zero in the
1127 * response parsing callback. */
1128 LogDebug( ( "http-parser errno description: %s",
1129 http_errno_description( HTTP_PARSER_ERRNO( pHttpParser ) ) ) );
1130
1131 return returnStatus;
1132 }

1136 static HTTPStatus_t parseHttpResponse( HTTPParsingContext_t * pParsingContext,
1137 HTTPResponse_t * pResponse,
1138 size_t parseLen )
1139 {
1140 HTTPStatus_t returnStatus;
1141 http_parser_settings parserSettings = { 0 };
1142 size_t bytesParsed = 0U;
1143 const char * parsingStartLoc = NULL;
1144
1145 /* Disable unused variable warning. */
1146 ( void ) bytesParsed;
1147
1148 assert( pParsingContext != NULL );
1149 assert( pResponse != NULL );
1150
1151 /* If this is the first time this parsing context is used, then set the
1152 * response input. */
1153 if( pParsingContext->pResponse == NULL )
1154 {
1155 pParsingContext->pResponse = pResponse;
1156 pParsingContext->pBufferCur = ( const char * ) pResponse->pBuffer;
1157
1158 /* Initialize the status-code returned in the response. */
1159 pResponse->statusCode = 0U;
1160 /* Initialize the start of the response body and length. */
1161 pResponse->pBody = NULL;
1162 pResponse->bodyLen = 0U;
1163
1164 /* Initialize the start of the headers, its length, and the count for
1165 * the parsing that follows the status. */
1166 pResponse->pHeaders = NULL;
1167 pResponse->headersLen = 0U;
1168 pResponse->headerCount = 0U;
1169 /* Initialize the response flags. */
1170 pResponse->respFlags = 0U;
1171 }
1172 else
1173 {
1174 /* This function is currently private to the HTTP Client library. It is
1175 * therefore a development bug to have this function invoked in
1176 * succession without the same response. */
1177 assert( pParsingContext->pResponse == pResponse );
1178 }
1179
1180 /* Initialize the callbacks that http_parser_execute will invoke. */
1181 http_parser_settings_init( &parserSettings );
1182 parserSettings.on_message_begin = httpParserOnMessageBeginCallback;
1183 parserSettings.on_status = httpParserOnStatusCallback;
1184 parserSettings.on_header_field = httpParserOnHeaderFieldCallback;
1185 parserSettings.on_header_value = httpParserOnHeaderValueCallback;
1186 parserSettings.on_headers_complete = httpParserOnHeadersCompleteCallback;
1187 parserSettings.on_body = httpParserOnBodyCallback;
1188 parserSettings.on_message_complete = httpParserOnMessageCompleteCallback;
1189
1190 /* Setting this allows the parsing context and response to be carried to
1191 * each of the callbacks that http_parser_execute() will invoke. */
1192 pParsingContext->httpParser.data = pParsingContext;
1193
1194 /* Save the starting response buffer location to parse. This is needed to
1195 * ensure that we move the next location to parse to exactly how many
1196 * characters were parsed in this call. */
1197 parsingStartLoc = pParsingContext->pBufferCur;
1198
1199 /* This will begin the parsing. Each of the callbacks set in
1200 * parserSettings will be invoked as parts of the HTTP response are
1201 * reached. */
1202 bytesParsed = http_parser_execute( &( pParsingContext->httpParser ),
1203 &parserSettings,
1204 parsingStartLoc,
1205 parseLen );
1206
1207 /* The next location to parse will always be after what has already
1208 * been parsed. */
1209 pParsingContext->pBufferCur = parsingStartLoc + bytesParsed;
1210
1211 LogDebug( ( "Parsed HTTP Response buffer: BytesParsed=%lu, "
1212 "ExpectedBytesParsed=%lu",
1213 ( unsigned long ) bytesParsed,
1214 ( unsigned long ) parseLen ) );
1215
1216 returnStatus = processHttpParserError( &( pParsingContext->httpParser ) );
1217
1218 return returnStatus;
1219 }

1223 static uint8_t convertInt32ToAscii( int32_t value,
1224 char * pBuffer,
1225 size_t bufferLength )
1226 {
1227 /* As input value may be altered and MISRA C 2012 rule 17.8 prevents
1228 * modification of parameter, a local copy of the parameter is stored.
1229 * absoluteValue stores the positive version of the input value. Its type
1230 * remains the same type as the input value to avoid unnecessary casting on
1231 * a privately used variable. This variable's size will always be less
1232 * than INT32_MAX. */
1233 int32_t absoluteValue = value;
1234 uint8_t numOfDigits = 0U;
1235 uint8_t index = 0U;
1236 uint8_t isNegative = 0U;
1237 char temp = '\0';
1238
1239 assert( pBuffer != NULL );
1240 assert( bufferLength >= MAX_INT32_NO_OF_DECIMAL_DIGITS );
1241 ( void ) bufferLength;
1242
1243 /* If the value is negative, write the '-' (minus) character to the buffer. */
1244 if( value < 0 )
1245 {
1246 isNegative = 1U;
1247
1248 *pBuffer = '-';
1249
1250 /* Convert the value to its absolute representation. */
1251 absoluteValue = value * -1;
1252 }
1253
1254 /* Write the absolute integer value in reverse ASCII representation. */
1255 do
1256 {
1257 pBuffer[ isNegative + numOfDigits ] = ( char ) ( ( absoluteValue % 10 ) + '0' );
1258 numOfDigits++;
1259 absoluteValue /= 10;
1260 } while( absoluteValue != 0 );
1261
1262 /* Reverse the digits in the buffer to store the correct ASCII representation
1263 * of the value. */
1264 for( index = 0U; index < ( numOfDigits / 2U ); index++ )
1265 {
1266 temp = pBuffer[ isNegative + index ];
1267 pBuffer[ isNegative + index ] = pBuffer[ isNegative + numOfDigits - index - 1U ];
1268 pBuffer[ isNegative + numOfDigits - index - 1U ] = temp;
1269 }
1270
1271 return( isNegative + numOfDigits );
1272 }

1309 {
1310 pDest[ i ] = pSrc[ i ];
1311 }
1312
1313 if( hasError == 1U )
1314 {

1317 }
1318 }

1325 static HTTPStatus_t addHeader( HTTPRequestHeaders_t * pRequestHeaders,
1326 const char * pField,
1327 size_t fieldLen,
1328 const char * pValue,
1329 size_t valueLen )
1330 {
1331 HTTPStatus_t returnStatus = HTTPSuccess;
1332 char * pBufferCur = NULL;
1333 size_t toAddLen = 0U;
1334 size_t backtrackHeaderLen = 0U;
1335
1336 assert( pRequestHeaders != NULL );
1337 assert( pRequestHeaders->pBuffer != NULL );
1338 assert( pField != NULL );
1339 assert( pValue != NULL );
1340 assert( fieldLen != 0U );
1341 assert( valueLen != 0U );
1342
1343 pBufferCur = ( char * ) ( pRequestHeaders->pBuffer + pRequestHeaders->headersLen );
1344 backtrackHeaderLen = pRequestHeaders->headersLen;
1345
1346 /* Backtrack before trailing "\r\n" (HTTP header end) if it's already written.
1347 * Note that this method also writes trailing "\r\n" before returning.
1348 * The first condition prevents reading before start of the header. */
1349 if( ( HTTP_HEADER_END_INDICATOR_LEN <= pRequestHeaders->headersLen ) &&
1350 ( strncmp( ( char * ) pBufferCur - HTTP_HEADER_END_INDICATOR_LEN,
1351 HTTP_HEADER_END_INDICATOR, HTTP_HEADER_END_INDICATOR_LEN ) == 0 ) )
1352 {
1353 backtrackHeaderLen -= HTTP_HEADER_LINE_SEPARATOR_LEN;
1354 pBufferCur -= HTTP_HEADER_LINE_SEPARATOR_LEN;
1355 }
1356
1357 /* Check if there is enough space in buffer for additional header. */
1358 toAddLen = fieldLen + HTTP_HEADER_FIELD_SEPARATOR_LEN + valueLen +
1359 HTTP_HEADER_LINE_SEPARATOR_LEN +
1360 HTTP_HEADER_LINE_SEPARATOR_LEN;
1361
1362 /* If we have enough room for the new header line, then write it to the
1363 * header buffer. */
1364 if( ( backtrackHeaderLen + toAddLen ) <= pRequestHeaders->bufferLen )
1365 {
1366 /* Write "<Field>: <Value> \r\n" to the headers buffer. */
1367
1368 /* Copy the header name into the buffer. */
1369 if( httpHeaderStrncpy( pBufferCur, pField, fieldLen, HTTP_HEADER_STRNCPY_IS_FIELD ) == NULL )
1370 {
1371 returnStatus = HTTPSecurityAlertInvalidCharacter;
1372 }
1373
1374 if( returnStatus == HTTPSuccess )
1375 {
1376 pBufferCur += fieldLen;
1377
1378 /* Copy the field separator, ": ", into the buffer. */
1379 ( void ) strncpy( pBufferCur,
1380 HTTP_HEADER_FIELD_SEPARATOR,
1381 HTTP_HEADER_FIELD_SEPARATOR_LEN );
1382
1383 pBufferCur += HTTP_HEADER_FIELD_SEPARATOR_LEN;
1384
1385 /* Copy the header value into the buffer. */
1386 if( httpHeaderStrncpy( pBufferCur, pValue, valueLen, HTTP_HEADER_STRNCPY_IS_VALUE ) == NULL )
1387 {
1388 returnStatus = HTTPSecurityAlertInvalidCharacter;
1389 }
1390 }
1391
1392 if( returnStatus == HTTPSuccess )
1393 {
1394 pBufferCur += valueLen;
1395
1396 /* Copy the header end indicator, "\r\n\r\n" into the buffer. */
1397 ( void ) strncpy( pBufferCur,
1398 HTTP_HEADER_END_INDICATOR,
1399 HTTP_HEADER_END_INDICATOR_LEN );
1400
1401 /* Update the headers length value only when everything is successful. */
1402 pRequestHeaders->headersLen = backtrackHeaderLen + toAddLen;
1403 }
1404 }
1405 else
1406 {
1407 LogError( ( "Unable to add header in buffer: "
1408 "Buffer has insufficient memory: "
1409 "RequiredBytes=%lu, RemainingBufferSize=%lu",
1410 ( unsigned long ) toAddLen,
1411 ( unsigned long ) ( pRequestHeaders->bufferLen - pRequestHeaders->headersLen ) ) );
1412 returnStatus = HTTPInsufficientMemory;
1413 }
1414
1415 return returnStatus;
1416 }

1419
1420 static HTTPStatus_t addRangeHeader( HTTPRequestHeaders_t * pRequestHeaders,
1421 int32_t rangeStartOrlastNbytes,
1422 int32_t rangeEnd )
1423 {
1424 HTTPStatus_t returnStatus = HTTPSuccess;
1425 char rangeValueBuffer[ HTTP_MAX_RANGE_REQUEST_VALUE_LEN ];
1426 size_t rangeValueLength = 0U;
1427
1428 assert( pRequestHeaders != NULL );
1429
1430 /* This buffer uses a char type instead of the general purpose uint8_t
1431 * because the range value expected to be written is within the ASCII
1432 * character set. */
1433 ( void ) memset( rangeValueBuffer, 0, HTTP_MAX_RANGE_REQUEST_VALUE_LEN );
1434
1435 /* Generate the value data for the Range Request header.*/
1436
1437 /* Write the range value prefix in the buffer. */
1438 ( void ) strncpy( rangeValueBuffer,
1439 HTTP_RANGE_REQUEST_HEADER_VALUE_PREFIX,
1440 HTTP_RANGE_REQUEST_HEADER_VALUE_PREFIX_LEN );
1441 rangeValueLength += HTTP_RANGE_REQUEST_HEADER_VALUE_PREFIX_LEN;
1442
1443 /* Write the range start value in the buffer. */
1444 rangeValueLength += convertInt32ToAscii( rangeStartOrlastNbytes,
1445 rangeValueBuffer + rangeValueLength,
1446 sizeof( rangeValueBuffer ) - rangeValueLength );
1447
1448 /* Add remaining value data depending on the range specification type. */
1449
1450 /* Add rangeEnd value if request is for [rangeStart, rangeEnd] byte range */
1451 if( rangeEnd != HTTP_RANGE_REQUEST_END_OF_FILE )
1452 {
1453 /* Write the "-" character to the buffer.*/
1454 *( rangeValueBuffer + rangeValueLength ) = DASH_CHARACTER;
1455 rangeValueLength += DASH_CHARACTER_LEN;
1456
1457 /* Write the rangeEnd value of the request range to the buffer. */
1458 rangeValueLength += convertInt32ToAscii( rangeEnd,
1459 rangeValueBuffer + rangeValueLength,
1460 sizeof( rangeValueBuffer ) - rangeValueLength );
1461 }
1462 /* Case when request is for bytes in the range [rangeStart, EoF). */
1463 else if( rangeStartOrlastNbytes >= 0 )
1464 {
1465 /* Write the "-" character to the buffer.*/
1466 *( rangeValueBuffer + rangeValueLength ) = DASH_CHARACTER;
1467 rangeValueLength += DASH_CHARACTER_LEN;
1468 }
1469 else
1470 {
1471 /* Empty else MISRA 15.7 */
1472 }
1473
1474 /* Add the Range Request header field and value to the buffer. */
1475 returnStatus = addHeader( pRequestHeaders,
1476 HTTP_RANGE_REQUEST_HEADER_FIELD,
1477 HTTP_RANGE_REQUEST_HEADER_FIELD_LEN,
1478 rangeValueBuffer,
1479 rangeValueLength );
1480
1481 return returnStatus;
1482 }

1486 static HTTPStatus_t writeRequestLine( HTTPRequestHeaders_t * pRequestHeaders,
1487 const char * pMethod,
1488 size_t methodLen,
1489 const char * pPath,
1490 size_t pathLen )
1491 {
1492 HTTPStatus_t returnStatus = HTTPSuccess;
1493 char * pBufferCur = NULL;
1494 size_t toAddLen = 0U;
1495
1496 assert( pRequestHeaders != NULL );
1497 assert( pRequestHeaders->pBuffer != NULL );
1498 assert( pMethod != NULL );
1499 assert( methodLen != 0U );
1500
1501 toAddLen = methodLen + \
1502 SPACE_CHARACTER_LEN + \
1503 SPACE_CHARACTER_LEN + \
1504 HTTP_PROTOCOL_VERSION_LEN + \
1505 HTTP_HEADER_LINE_SEPARATOR_LEN;
1506
1507 pBufferCur = ( char * ) ( pRequestHeaders->pBuffer );
1508 toAddLen += ( ( pPath == NULL ) || ( pathLen == 0U ) ) ? HTTP_EMPTY_PATH_LEN : pathLen;
1509
1510 if( ( toAddLen + pRequestHeaders->headersLen ) > pRequestHeaders->bufferLen )
1511 {
1512 returnStatus = HTTPInsufficientMemory;
1513 }
1514
1515 if( returnStatus == HTTPSuccess )
1516 {
1517 /* Write "<METHOD> <PATH> HTTP/1.1\r\n" to start the HTTP header. */
1518 ( void ) strncpy( pBufferCur, pMethod, methodLen );
1519 pBufferCur += methodLen;
1520
1521 *pBufferCur = SPACE_CHARACTER;
1522 pBufferCur += SPACE_CHARACTER_LEN;
1523
1524 /* Use "/" as default value if <PATH> is NULL. */
1525 if( ( pPath == NULL ) || ( pathLen == 0U ) )
1526 {
1527 ( void ) strncpy( pBufferCur,
1528 HTTP_EMPTY_PATH,
1529 HTTP_EMPTY_PATH_LEN );
1530 pBufferCur += HTTP_EMPTY_PATH_LEN;
1531 }
1532 else
1533 {
1534 ( void ) strncpy( pBufferCur, pPath, pathLen );
1535 pBufferCur += pathLen;
1536 }
1537
1538 *pBufferCur = SPACE_CHARACTER;
1539 pBufferCur += SPACE_CHARACTER_LEN;
1540
1541 ( void ) strncpy( pBufferCur,
1542 HTTP_PROTOCOL_VERSION,
1543 HTTP_PROTOCOL_VERSION_LEN );
1544 pBufferCur += HTTP_PROTOCOL_VERSION_LEN;
1545
1546 ( void ) strncpy( pBufferCur,
1547 HTTP_HEADER_LINE_SEPARATOR,
1548 HTTP_HEADER_LINE_SEPARATOR_LEN );
1549 pRequestHeaders->headersLen = toAddLen;
1550 }
1551
1552 return returnStatus;
1553 }

1779 static HTTPStatus_t sendHttpData( const TransportInterface_t * pTransport,
1780 HTTPClient_GetCurrentTimeFunc_t getTimestampMs,
1781 const uint8_t * pData,
1782 size_t dataLen )
1783 {
1784 HTTPStatus_t returnStatus = HTTPSuccess;
1785 const uint8_t * pIndex = pData;
1786 int32_t bytesSent = 0;
1787 size_t bytesRemaining = dataLen;
1788 uint32_t lastSendTimeMs = 0U, timeSinceLastSendMs = 0U;
1789 uint32_t retryTimeoutMs = HTTP_SEND_RETRY_TIMEOUT_MS;
1790
1791 assert( pTransport != NULL );
1792 assert( pTransport->send != NULL );
1793 assert( pData != NULL );
1794
1795 /* If the timestamp function was undefined by the application, then do not
1796 * retry the transport send. */
1797 if( getTimestampMs == getZeroTimestampMs )
1798 {
1799 retryTimeoutMs = 0U;
1800 }
1801
1802 /* Initialize the last send time to allow retries, if 0 bytes are sent on
1803 * the first try. */
1804 lastSendTimeMs = getTimestampMs();
1805
1806 /* Loop until all data is sent. */
1807 while( ( bytesRemaining > 0UL ) && ( returnStatus != HTTPNetworkError ) )
1808 {
1809 bytesSent = pTransport->send( pTransport->pNetworkContext,
1810 pIndex,
1811 bytesRemaining );
1812
1813 /* BytesSent less than zero is an error. */
1814 if( bytesSent < 0 )
1815 {
1816 LogError( ( "Failed to send data: Transport send error: "
1817 "TransportStatus=%ld", ( long int ) bytesSent ) );
1818 returnStatus = HTTPNetworkError;
1819 }
1820 else if( bytesSent > 0 )
1821 {
1822 /* It is a bug in the application's transport send implementation if
1823 * more bytes than expected are sent. To avoid a possible overflow
1824 * in converting bytesRemaining from unsigned to signed, this assert
1825 * must exist after the check for bytesSent being negative. */
1826 assert( ( size_t ) bytesSent <= bytesRemaining );
1827
1828 /* Record the most recent time of successful transmission. */
1829 lastSendTimeMs = getTimestampMs();
1830
1831 bytesRemaining -= ( size_t ) bytesSent;
1832 pIndex += bytesSent;
1833 LogDebug( ( "Sent data over the transport: "
1834 "BytesSent=%ld, BytesRemaining=%lu, TotalBytesSent=%lu",
1835 ( long int ) bytesSent,
1836 ( unsigned long ) bytesRemaining,
1837 ( unsigned long ) ( dataLen - bytesRemaining ) ) );
1838 }
1839 else
1840 {
1841 /* No bytes were sent over the network. */
1842 timeSinceLastSendMs = getTimestampMs() - lastSendTimeMs;
1843
1844 /* Check for timeout if we have been waiting to send any data over
1845 * the network. */
1846 if( timeSinceLastSendMs >= retryTimeoutMs )
1847 {
1848 LogError( ( "Unable to send packet: Timed out in transport send." ) );
1849 returnStatus = HTTPNetworkError;
1850 }
1851 }
1852 }
1853
1854 return returnStatus;
1855 }

1996
1997 static HTTPStatus_t receiveAndParseHttpResponse( const TransportInterface_t * pTransport,
1998 HTTPResponse_t * pResponse,
1999 const HTTPRequestHeaders_t * pRequestHeaders )
2000 {
2001 HTTPStatus_t returnStatus = HTTPSuccess;
2002 size_t totalReceived = 0U;
2003 int32_t currentReceived = 0;
2004 HTTPParsingContext_t parsingContext = { 0 };
2005 uint8_t shouldRecv = 1U, shouldParse = 1U, timeoutReached = 0U;
2006 uint32_t lastRecvTimeMs = 0U, timeSinceLastRecvMs = 0U;
2007 uint32_t retryTimeoutMs = HTTP_RECV_RETRY_TIMEOUT_MS;
2008
2009 assert( pTransport != NULL );
2010 assert( pTransport->recv != NULL );
2011 assert( pResponse != NULL );
2012 assert( pRequestHeaders != NULL );
2013
2014 /* Initialize the parsing context for parsing the response received from the
2015 * network. */
2016 initializeParsingContextForFirstResponse( &parsingContext, pRequestHeaders );
2017
2018 /* If the timestamp function was undefined by the application, then do not
2019 * retry the transport receive. */
2020 if( pResponse->getTime == getZeroTimestampMs )
2021 {
2022 retryTimeoutMs = 0U;
2023 }
2024
2025 /* Initialize the last send time to allow retries, if 0 bytes are sent on
2026 * the first try. */
2027 lastRecvTimeMs = pResponse->getTime();
2028
2029 while( shouldRecv == 1U )
2030 {
2031 /* Receive the HTTP response data into the pResponse->pBuffer. */
2032 currentReceived = pTransport->recv( pTransport->pNetworkContext,
2033 pResponse->pBuffer + totalReceived,
2034 pResponse->bufferLen - totalReceived );
2035
2036 /* Transport receive errors are negative. */
2037 if( currentReceived < 0 )
2038 {
2039 LogError( ( "Failed to receive HTTP data: Transport recv() "
2040 "returned error: TransportStatus=%ld",
2041 ( long int ) currentReceived ) );
2042 returnStatus = HTTPNetworkError;
2043
2044 /* Do not invoke the parser on network errors. */
2045 shouldParse = 0U;
2046 }
2047 else if( currentReceived > 0 )
2048 {
2049 /* Reset the time of the last data received when data is received. */
2050 lastRecvTimeMs = pResponse->getTime();
2051
2052 /* Parsing is done on data as soon as it is received from the network.
2053 * Because we cannot know how large the HTTP response will be in
2054 * total, parsing will tell us if the end of the message is reached.*/
2055 shouldParse = 1U;
2056 totalReceived += currentReceived;
2057 }
2058 else
2059 {
2060 timeSinceLastRecvMs = pResponse->getTime() - lastRecvTimeMs;
2061 /* Do not invoke the response parsing for intermediate zero data. */
2062 shouldParse = 0U;
2063
2064 /* Check if the allowed elapsed time between non-zero data has been
2065 * reached. */
2066 if( timeSinceLastRecvMs >= retryTimeoutMs )
2067 {
2068 /* Invoke the parsing upon this final zero data to indicate
2069 * to the parser that there is no more data available from the
2070 * server. */
2071 shouldParse = 1U;
2072 timeoutReached = 1U;
2073 }
2074 }
2075
2076 if( shouldParse == 1U )
2077 {
2078 /* Data is received into the buffer is immediately parsed. Parsing
2079 * is invoked even with a length of zero. A length of zero indicates
2080 * to the parser that there is no more data from the server (EOF). */
2081 returnStatus = parseHttpResponse( &parsingContext,
2082 pResponse,
2083 currentReceived );
2084 }
2085
2086 /* Reading should continue if there are no errors in the transport receive
2087 * or parsing, the retry on zero data timeout has not been reached, the
2088 * parser indicated the response message is not finished, and there is
2089 * room in the response buffer. */
2090 shouldRecv = ( ( returnStatus == HTTPSuccess ) &&
2091 ( timeoutReached == 0U ) &&
2092 ( parsingContext.state != HTTP_PARSING_COMPLETE ) &&
2093 ( totalReceived < pResponse->bufferLen ) ) ? 1U : 0U;
2094 }
2095
2096 if( returnStatus == HTTPSuccess )
2097 {
2098 /* If there are errors in receiving from the network or during parsing,
2099 * the final status of the response message is derived from the state of
2100 * the parsing and how much data is in the buffer. */
2101 returnStatus = getFinalResponseStatus( parsingContext.state,
2102 totalReceived,
2103 pResponse->bufferLen );
2104 }
2105
2106 return returnStatus;
2107 }

2380 static HTTPStatus_t findHeaderInResponse( const uint8_t * pBuffer,
2381 size_t bufferLen,
2382 const char * pField,
2383 size_t fieldLen,
2384 const char ** pValueLoc,
2385 size_t * pValueLen )
2386 {
2387 HTTPStatus_t returnStatus = HTTPSuccess;
2388 http_parser parser = { 0 };
2389 http_parser_settings parserSettings = { 0 };
2390 findHeaderContext_t context = { 0 };
2391 size_t numOfBytesParsed = 0U;
2392
2393 context.pField = pField;
2394 context.fieldLen = fieldLen;
2395 context.pValueLoc = pValueLoc;
2396 context.pValueLen = pValueLen;
2397 context.fieldFound = 0U;
2398 context.valueFound = 0U;
2399
2400 /* Disable unused variable warning. This variable is used only in logging. */
2401 ( void ) numOfBytesParsed;
2402
2403 http_parser_init( &parser, HTTP_RESPONSE );
2404
2405 /* Set the context for the parser. */
2406 parser.data = &context;
2407
2408 /* The intention here to define callbacks just for searching the headers. We will
2409 * need to create a private context in httpParser->data that has the field and
2410 * value to update and pass back. */
2411 http_parser_settings_init( &parserSettings );
2412 parserSettings.on_header_field = findHeaderFieldParserCallback;
2413 parserSettings.on_header_value = findHeaderValueParserCallback;
2414 parserSettings.on_headers_complete = findHeaderOnHeaderCompleteCallback;
2415
2416 /* Start parsing for the header! */
2417 numOfBytesParsed = http_parser_execute( &parser,
2418 &parserSettings,
2419 ( const char * ) pBuffer,
2420 bufferLen );
2421
2422 LogDebug( ( "Parsed response for header search: NumBytesParsed=%lu",
2423 ( unsigned long ) numOfBytesParsed ) );
2424
2425 if( context.fieldFound == 0U )
2426 {
2427 /* If header field is not found, then both the flags should be zero. */
2428 assert( context.valueFound == 0U );
2429
2430 /* Header is not present in buffer. */
2431 LogWarn( ( "Header not found in response buffer: RequestedHeader=%.*s",
2432 ( int ) fieldLen,
2433 pField ) );
2434
2435 returnStatus = HTTPHeaderNotFound;
2436 }
2437 else if( context.valueFound == 0U )
2438 {
2439 /* The response buffer is invalid as only the header field was found
2440 * in the "<field>: <value>\r\n" format of an HTTP header. */
2441 LogError( ( "Unable to find header value in response: "
2442 "Response data is invalid: "
2443 "RequestedHeader=%.*s, ParserError=%s",
2444 ( int ) fieldLen,
2445 pField,
2446 http_errno_description( HTTP_PARSER_ERRNO( &( parser ) ) ) ) );
2447 returnStatus = HTTPInvalidResponse;
2448 }
2449 else
2450 {
2451 /* Header is found. */
2452 assert( ( context.fieldFound == 1U ) && ( context.valueFound == 1U ) );
2453
2454 LogDebug( ( "Found requested header in response: "
2455 "HeaderName=%.*s, HeaderValue=%.*s",
2456 ( int ) fieldLen,
2457 pField,
2458 ( int ) ( *pValueLen ),
2459 *pValueLoc ) );
2460 returnStatus = HTTPSuccess;
2461 }
2462
2463 /* If the header field-value pair is found in response, then the return
2464 * value of "on_header_value" callback (related to the header value) should
2465 * cause the http_parser.http_errno to be "CB_header_value". */
2466 if( ( returnStatus == HTTPSuccess ) &&
2467 ( parser.http_errno != ( unsigned int ) HPE_CB_header_value ) )
2468 {
2469 LogError( ( "Header found in response but http-parser returned error: "
2470 "ParserError=%s",
2471 http_errno_description( HTTP_PARSER_ERRNO( &( parser ) ) ) ) );
2472 returnStatus = HTTPParserInternalError;
2473 }
2474
2475 /* If header was not found, then the "on_header_complete" callback is
2476 * expected to be called which should cause the http_parser.http_errno to be
2477 * "OK" */
2478 else if( ( returnStatus == HTTPHeaderNotFound ) &&
2479 ( parser.http_errno != ( unsigned int ) ( HPE_OK ) ) )
2480 {
2481 LogError( ( "Header not found in response: http-parser returned error: "
2482 "ParserError=%s",
2483 http_errno_description( HTTP_PARSER_ERRNO( &( parser ) ) ) ) );
2484 returnStatus = HTTPInvalidResponse;
2485 }
2486 else
2487 {
2488 /* Empty else for MISRA 15.7 compliance. */
2489 }
2490
2491 return returnStatus;
2492 }
