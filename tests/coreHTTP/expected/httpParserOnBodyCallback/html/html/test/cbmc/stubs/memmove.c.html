23 /**
24 * @file memmove.c
25 * @brief Creates a stub for memmove so that the proof for
26 * httpParserOnBodyCallback is not stuck in the Post-Processing Step.
27 */
28
29 #include <string.h>
30
31 /* This is a clang macro not available on linux */
32 #ifndef __has_builtin
33 #define __has_builtin( x ) 0
34 #endif
35
36 #if __has_builtin( __builtin___memmove_chk )
37 void * __builtin___memmove_chk( void * dest,
38 const void * src,
39 size_t n,
40 size_t m )
41 {
42 __CPROVER_assert( __CPROVER_w_ok( dest, n ), "write" );
43 __CPROVER_assert( __CPROVER_r_ok( src, n ), "read" );
44 return dest;
45 }
46 #else
47 void * memmove( void * dest,
48 const void * src,
49 size_t n )
50 {
51 __CPROVER_assert( __CPROVER_w_ok( dest, n ), "write" );
52 __CPROVER_assert( __CPROVER_r_ok( src, n ), "read" );
53 return dest;
54 }
55 #endif /* if __has_builtin( __builtin___memmove_chk ) */
