40 /**
41 * @transportpage
42 * @brief The transport interface definition.
43 *
44 * @transportsectionoverview
45 *
46 * The transport interface is a set of APIs that must be implemented using an
47 * external transport layer protocol. The transport interface is defined in
48 * @ref transport_interface.h. This interface allows protocols like MQTT and
49 * HTTP to send and receive data over the transport layer. This
50 * interface does not handle connection and disconnection to the server of
51 * interest. The connection, disconnection, and other transport settings, like
52 * timeout and TLS setup, must be handled in the user application.
53 * <br>
54 *
55 * The functions that must be implemented are:<br>
56 * - [Transport Receive](@ref TransportRecv_t)
57 * - [Transport Send](@ref TransportSend_t)
58 *
59 * Each of the functions above take in an opaque context @ref NetworkContext_t.
60 * The functions above and the context are also grouped together in the
61 * @ref TransportInterface_t structure:<br><br>
62 * @snippet this define_transportinterface
63 * <br>
64 *
65 * @transportsectionimplementation
66 *
67 * The following steps give guidance on implementing the transport interface:
68 *
69 * -# Implementing @ref NetworkContext_t<br><br>
70 * @snippet this define_networkcontext
71 * <br>
72 * @ref NetworkContext_t is the incomplete type <b>struct NetworkContext</b>.
73 * The implemented struct NetworkContext must contain all of the information
74 * that is needed to receive and send data with the @ref TransportRecv_t
75 * and the @ref TransportSend_t implementations.<br>
76 * In the case of TLS over TCP, struct NetworkContext is typically implemented
77 * with the TCP socket context and a TLS context.<br><br>
78 * <b>Example code:</b>
79 * @code{c}
80 * struct NetworkContext
81 * {
82 * struct MyTCPSocketContext tcpSocketContext;
83 * struct MyTLSContext tlsContext;
84 * };
85 * @endcode
86 * <br>
87 * -# Implementing @ref TransportRecv_t<br><br>
88 * @snippet this define_transportrecv
89 * <br>
90 * This function is expected to populate a buffer, with bytes received from the
91 * transport, and return the number of bytes placed in the buffer.
92 * In the case of TLS over TCP, @ref TransportRecv_t is typically implemented by
93 * calling the TLS layer function to receive data. In case of plaintext TCP
94 * without TLS, it is typically implemented by calling the TCP layer receive
95 * function. @ref TransportRecv_t may be invoked multiple times by the protocol
96 * library, if fewer bytes than were requested to receive are returned.
97 * <br><br>
98 * <b>Example code:</b>
99 * @code{c}
100 * int32_t myNetworkRecvImplementation( NetworkContext_t * pNetworkContext,
101 * void * pBuffer,
102 * size_t bytesToRecv )
103 * {
104 * int32_t bytesReceived = 0;
105 * bool callTlsRecvFunc = true;
106 *
107 * // For a single byte read request, check if data is available on the network.
108 * if( bytesToRecv == 1 )
109 * {
110 * // If no data is available on the network, do not call TLSRecv
111 * // to avoid blocking for socket timeout.
112 * if( TLSRecvCount( pNetworkContext->tlsContext ) == 0 )
113 * {
114 * callTlsRecvFunc = false;
115 * }
116 * }
117 *
118 * if( callTlsRecvFunc == true )
119 * {
120 * bytesReceived = TLSRecv( pNetworkContext->tlsContext,
121 * pBuffer,
122 * bytesToRecv,
123 * MY_SOCKET_TIMEOUT );
124 * if( bytesReceived < 0 )
125 * {
126 * // If the error code represents a timeout, then the return
127 * // code should be translated to zero so that the caller
128 * // can retry the read operation.
129 * if( bytesReceived == MY_SOCKET_ERROR_TIMEOUT )
130 * {
131 * bytesReceived = 0;
132 * }
133 * }
134 * // Handle other cases.
135 * }
136 * return bytesReceived;
137 * }
138 * @endcode
139 * <br>
140 * -# Implementing @ref TransportSend_t<br><br>
141 * @snippet this define_transportsend
142 * <br>
143 * This function is expected to send the bytes, in the given buffer over the
144 * transport, and return the number of bytes sent.
145 * In the case of TLS over TCP, @ref TransportSend_t is typically implemented by
146 * calling the TLS layer function to send data. In case of plaintext TCP
147 * without TLS, it is typically implemented by calling the TCP layer send
148 * function. @ref TransportSend_t may be invoked multiple times by the protocol
149 * library, if fewer bytes than were requested to send are returned.
150 * <br><br>
151 * <b>Example code:</b>
152 * @code{c}
153 * int32_t myNetworkSendImplementation( NetworkContext_t * pNetworkContext,
154 * const void * pBuffer,
155 * size_t bytesToSend )
156 * {
157 * int32_t bytesSent = 0;
158 * bytesSent = TLSSend( pNetworkContext->tlsContext,
159 * pBuffer,
160 * bytesToSend,
161 * MY_SOCKET_TIMEOUT );
162 *
163 * // If underlying TCP buffer is full, set the return value to zero
164 * // so that caller can retry the send operation.
165 * if( bytesSent == MY_SOCKET_ERROR_BUFFER_FULL )
166 * {
167 * bytesSent = 0;
168 * }
169 * else if( bytesSent < 0 )
170 * {
171 * // Handle socket error.
172 * }
173 * // Handle other cases.
174 *
175 * return bytesSent;
176 * }
177 * @endcode
178 */
179
180 /**
181 * @transportstruct
182 * @typedef NetworkContext_t
183 * @brief The NetworkContext is an incomplete type. An implementation of this
184 * interface must define struct NetworkContext for the system requirements.
185 * This context is passed into the network interface functions.
186 */
187 /* @[define_networkcontext] */
188 struct NetworkContext;
189 typedef struct NetworkContext NetworkContext_t;
190 /* @[define_networkcontext] */
191
192 /**
193 * @transportcallback
194 * @brief Transport interface for receiving data on the network.
195 *
196 * @note It is RECOMMENDED that the transport receive implementation
197 * does NOT block when requested to read a single byte. A single byte
198 * read request can be made by the caller to check whether there is a
199 * new frame available on the network for reading.
200 * However, the receive implementation MAY block for a timeout period when
201 * it is requested to read more than 1 byte. This is because once the caller
202 * is aware that a new frame is available to read on the network, then
203 * the likelihood of reading more than one byte over the network becomes high.
204 *
205 * @param[in] pNetworkContext Implementation-defined network context.
206 * @param[in] pBuffer Buffer to receive the data into.
207 * @param[in] bytesToRecv Number of bytes requested from the network.
208 *
209 * @return The number of bytes received or a negative value to indicate
210 * error.
211 *
212 * @note If no data is available on the network to read and no error
213 * has occurred, zero MUST be the return value. A zero return value
214 * SHOULD represent that the read operation can be retried by calling
215 * the API function. Zero MUST NOT be returned if a network disconnection
216 * has occurred.
217 */
218 /* @[define_transportrecv] */
219 typedef int32_t ( * TransportRecv_t )( NetworkContext_t * pNetworkContext,
220 void * pBuffer,
221 size_t bytesToRecv );
222 /* @[define_transportrecv] */
223
224 /**
225 * @transportcallback
226 * @brief Transport interface for sending data over the network.
227 *
228 * @param[in] pNetworkContext Implementation-defined network context.
229 * @param[in] pBuffer Buffer containing the bytes to send over the network stack.
230 * @param[in] bytesToSend Number of bytes to send over the network.
231 *
232 * @return The number of bytes sent or a negative value to indicate error.
233 *
234 * @note If no data is transmitted over the network due to a full TX buffer and
235 * no network error has occurred, this MUST return zero as the return value.
236 * A zero return value SHOULD represent that the send operation can be retried
237 * by calling the API function. Zero MUST NOT be returned if a network disconnection
238 * has occurred.
239 */
240 /* @[define_transportsend] */
241 typedef int32_t ( * TransportSend_t )( NetworkContext_t * pNetworkContext,
242 const void * pBuffer,
243 size_t bytesToSend );
244 /* @[define_transportsend] */
245
246 /**
247 * @transportstruct
248 * @brief The transport layer interface.
249 */
250 /* @[define_transportinterface] */
251 typedef struct TransportInterface
252 {
253 TransportRecv_t recv; /**< Transport receive interface. */
254 TransportSend_t send; /**< Transport send interface. */
255 NetworkContext_t * pNetworkContext; /**< Implementation-defined network context. */
256 } TransportInterface_t;
257 /* @[define_transportinterface] */
