23 /**
24 * @file findHeaderFieldParserCallback_harness.c
25 * @brief Implements the proof harness for findHeaderFieldParserCallback function.
26 */
27
28 #include "http_cbmc_state.h"
29 #include "http_parser.h"
30 #include "core_http_client.h"
31
32
33 void findHeaderFieldParserCallback_harness()
34 {
35 http_parser * pHttpParser;
36 HTTPResponse_t * pResponse;
37 findHeaderContext_t * pFindHeaderContext;
38 size_t fieldLen, fieldContextLen, fieldOffset, valueLen, valueOffset;
39 char * pFieldLoc;
40
41 pHttpParser = allocateHttpReadHeaderParser( NULL );
42 pFindHeaderContext = allocateFindHeaderContext( NULL );
43 pResponse = allocateHttpResponse( NULL );
44 __CPROVER_assume( isValidHttpResponse( pResponse ) &&
45 pResponse != NULL &&
46 pResponse->pBuffer != NULL &&
47 pResponse->bufferLen > 0 );
48
49 __CPROVER_assume( 0 < fieldLen && fieldLen < MAX_HEADER_FIELD_LENGTH && fieldLen <= pResponse->bufferLen );
50 __CPROVER_assume( fieldOffset <= pResponse->bufferLen - fieldLen );
51 pFieldLoc = pResponse->pBuffer + fieldOffset;
52
53 __CPROVER_assume( 0 < fieldContextLen && fieldContextLen < CBMC_MAX_OBJECT_SIZE );
54 pFindHeaderContext->pField = ( char * ) malloc( fieldContextLen );
55 __CPROVER_assume( pFindHeaderContext->pField != NULL );
56 pFindHeaderContext->fieldLen = fieldContextLen;
57 pFindHeaderContext->fieldFound = 0;
58 pFindHeaderContext->valueFound = 0;
59 pHttpParser->data = ( void * ) pFindHeaderContext;
60
61 __CPROVER_file_local_core_http_client_c_findHeaderFieldParserCallback( pHttpParser, pFieldLoc, fieldLen );
62 }
