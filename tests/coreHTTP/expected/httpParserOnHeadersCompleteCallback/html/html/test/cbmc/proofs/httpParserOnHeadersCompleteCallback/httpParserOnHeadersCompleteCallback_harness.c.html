23 /**
24 * @file httpParserOnHeadersCompleteCallback_harness.c
25 * @brief Implements the proof harness for httpParserOnHeadersCompleteCallback function.
26 */
27
28 #include "http_cbmc_state.h"
29 #include "http_parser.h"
30 #include "callback_stubs.h"
31
32
33 void httpParserOnHeadersCompleteCallback_harness()
34 {
35 http_parser * pHttpParser;
36 HTTPParsingContext_t * pParsingContext;
37 HTTPResponse_t * pResponse;
38 HTTPClient_ResponseHeaderParsingCallback_t headerParserCallback;
39 size_t bufferOffset;
40 size_t headerOffSet;
41
42 pHttpParser = allocateHttpSendParser( NULL );
43
44 pParsingContext = ( HTTPParsingContext_t * ) ( pHttpParser->data );
45 headerParserCallback.onHeaderCallback = onHeaderCallbackStub;
46
47 pResponse = pParsingContext->pResponse;
48 pResponse->pHeaderParsingCallback = &headerParserCallback;
49
50 /* The next place to parse must be between the start of the buffer and the
51 * end of the buffer. This assumption is an assertion in the function. */
52 __CPROVER_assume( bufferOffset < pResponse->bufferLen );
53 pParsingContext->pBufferCur = pResponse->pBuffer + bufferOffset;
54
55 /* The response headers MUST have been set prior or be NULL. This assumption
56 * is an assertion in the function. */
57 if( pResponse->pHeaders != NULL )
58 {
59 __CPROVER_assume( headerOffSet < bufferOffset );
60 pResponse->pHeaders = pResponse->pBuffer + headerOffSet;
61 }
62
63 /* This assumption suppresses an overflow error when incrementing pResponse->headerCount. */
64 __CPROVER_assume( pResponse->headerCount < SIZE_MAX );
65
66 __CPROVER_file_local_core_http_client_c_httpParserOnHeadersCompleteCallback( pHttpParser );
67 }
