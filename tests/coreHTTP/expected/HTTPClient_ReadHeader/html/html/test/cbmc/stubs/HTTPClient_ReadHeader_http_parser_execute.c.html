25 * @brief A stub function for http_parser_execute for coverage of
26 * #HTTPClient_ReadHeader.
27 */
28
29 #include <stdbool.h>
30 #include <stdint.h>
31
32 #include "core_http_client.h"
33 #include "core_http_client_private.h"
34 #include "http_parser.h"
35
36 /**
37 * @brief Attains coverage when a variable needs to possibly contain two values.
38 */
39 bool nondet_bool();
40
41 size_t http_parser_execute( http_parser * parser,
42 const http_parser_settings * settings,
43 const char * data,
44 size_t len )
45 {
46 char * pValue;
47 size_t fieldLength, fieldOffset, valueLength, valueOffset;
48 unsigned int http_errno;
49 findHeaderContext_t * pParsingContext;
50
51 __CPROVER_assert( parser != NULL,
52 "http_parser_execute parser is NULL" );
53 __CPROVER_assert( settings != NULL,
54 "http_parser_execute settings is NULL" );
55 __CPROVER_assert( data != NULL,
56 "http_parser_execute data is NULL" );
57 __CPROVER_assert( len < CBMC_MAX_OBJECT_SIZE,
58 "http_parser_execute len >= CBMC_MAX_OBJECT_SIZE" );
59
60 parser->http_errno = http_errno;
61
62 __CPROVER_assume( fieldLength <= len );
63 __CPROVER_assume( fieldOffset < fieldLength );
64 __CPROVER_assume( valueLength <= len );
65 __CPROVER_assume( valueOffset < valueLength );
66
67 pParsingContext = ( findHeaderContext_t * ) ( parser->data );
68 pParsingContext->pField = data + fieldOffset;
69 pParsingContext->fieldLen = fieldLength;
70 pParsingContext->pValueLoc = NULL;
71 pParsingContext->pValueLen = 0;
72 pParsingContext->fieldFound = nondet_bool() ? 0 : 1;
73
74 if( pParsingContext->fieldFound )
75 {
76 pParsingContext->valueFound = nondet_bool() ? 0 : 1;
77 }
78 else
79 {
80 pParsingContext->valueFound = 0;
81 }
82
83 if( pParsingContext->valueFound )
84 {
85 pValue = data + valueOffset;
86 pParsingContext->pValueLen = &valueLength;
87 pParsingContext->pValueLoc = &pValue;
88 }
89
90 return pParsingContext->fieldFound ? valueLength : 0;
91 }
