23 /**
24 * @file HTTPClient_ReadHeader_harness.c
25 * @brief Implements the proof harness for HTTPClient_ReadHeader function.
26 */
27
28 #include "core_http_client.h"
29
30 #include "http_cbmc_state.h"
31
32 void HTTPClient_ReadHeader_harness()
33 {
34 HTTPResponse_t * pResponse;
35 char * pField;
36 char ** pValue;
37 size_t fieldLen;
38 size_t valueLen;
39
40 /* Initialize and make assumptions for header field. */
41 __CPROVER_assume( fieldLen < CBMC_MAX_OBJECT_SIZE );
42 pField = mallocCanFail( fieldLen );
43
44 /* Initialize and make assumptions for header value. */
45 __CPROVER_assume( valueLen < CBMC_MAX_OBJECT_SIZE );
46 pValue = mallocCanFail( sizeof( char * ) );
47
48 /* Initialize and make assumptions for response object. */
49 pResponse = allocateHttpResponse( NULL );
50 __CPROVER_assume( isValidHttpResponse( pResponse ) );
51
52 HTTPClient_ReadHeader( pResponse,
53 pField,
54 fieldLen,
55 pValue,
56 valueLen );
57 }
