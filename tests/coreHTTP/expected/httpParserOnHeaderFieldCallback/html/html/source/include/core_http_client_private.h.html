164 /**
165 * @brief The minimum request-line in the headers has a possible one character
166 * custom method and a single forward / or asterisk * for the path:
167 *
168 * @code
169 * <1 character custom method> <1 character / or *> HTTP/1.x\r\n\r\n
170 * @endcode
171 *
172 * Therefore the minimum length is 16. If this minimum request-line is not
173 * satisfied, then the request headers to send are invalid.
174 *
175 * Note that custom methods are allowed per:
176 * https://tools.ietf.org/html/rfc2616#section-5.1.1.
177 */
178 #define HTTP_MINIMUM_REQUEST_LINE_LENGTH 16u

180 /**
181 * @brief The state of the response message parsed after function
182 * #parseHttpResponse returns.
183 */
184 typedef enum HTTPParsingState_t
185 {
186 HTTP_PARSING_NONE = 0, /**< The parser has not started reading any response. */
187 HTTP_PARSING_INCOMPLETE, /**< The parser found a partial reponse. */
188 HTTP_PARSING_COMPLETE /**< The parser found the entire response. */
189 } HTTPParsingState_t;

191 /**

196 typedef struct findHeaderContext

203 uint8_t valueFound; /**< Indicates that the header value was found during parsing. */
204 } findHeaderContext_t;

206 /**
207 * @brief The HTTP response parsing context for a response fresh from the
208 * server. This context is passed into the http-parser registered callbacks.
209 * The registered callbacks are private functions of the form
210 * httpParserXXXXCallbacks().
211 *
212 * The transitions of the httpParserXXXXCallback() functions are shown below.
213 * The XXXX is replaced by the strings in the state boxes:
214 *
215 * +---------------------+
216 * |onMessageBegin |
217 * +--------+------------+
218 * |
219 * |
220 * |
221 * v
222 * +--------+------------+
223 * |onStatus |
224 * +--------+------------+
225 * |
226 * |
227 * |
228 * v
229 * +--------+------------+
230 * |onHeaderField +<---+
231 * +--------+------------+ |
232 * | |
233 * | |(More headers)
234 * | |
235 * v |
236 * +--------+------------+ |
237 * |onHeaderValue +----^
238 * +--------+------------+
239 * |
240 * |
241 * |
242 * v
243 * +--------+------------+
244 * |onHeadersComplete |
245 * +---------------------+
246 * |
247 * |
248 * |
249 * v
250 * +--------+------------+
251 * |onBody +<---+
252 * +--------+--------+---+ |
253 * | | |(Transfer-encoding chunked body)
254 * | | |
255 * | +--------+
256 * |
257 * v
258 * +--------+------------+
259 * |onMessageComplete |
260 * +---------------------+
261 */
262 typedef struct HTTPParsingContext
263 {
264 http_parser httpParser; /**< Third-party http-parser context. */
265 HTTPParsingState_t state; /**< The current state of the HTTP response parsed. */
266 HTTPResponse_t * pResponse; /**< HTTP response associated with this parsing context. */
267 uint8_t isHeadResponse; /**< HTTP response is for a HEAD request. */
268
269 const char * pBufferCur; /**< The current location of the parser in the response buffer. */
270 const char * pLastHeaderField; /**< Holds the last part of the header field parsed. */
271 size_t lastHeaderFieldLen; /**< The length of the last header field parsed. */
272 const char * pLastHeaderValue; /**< Holds the last part of the header value parsed. */
273 size_t lastHeaderValueLen; /**< The length of the last value field parsed. */
274 } HTTPParsingContext_t;
