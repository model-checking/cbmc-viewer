44 /**

49 /**
50 * @brief Calls malloc based on given size or returns NULL for coverage.
51 *
52 * Implementation of safe malloc which returns NULL if the requested size is 0.
53 * The behavior of malloc(0) is platform dependent.
54 * It is possible for malloc(0) to return an address without allocating memory.
55 *
56 * @param[in] size Requested size to malloc.
57 *
58 * @return Requested memory or NULL.
59 */
60 void * mallocCanFail( size_t size );
61

68 */
