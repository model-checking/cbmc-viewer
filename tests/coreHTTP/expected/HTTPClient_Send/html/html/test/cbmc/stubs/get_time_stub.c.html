23 /**
24 * @file get_time_stub.c
25 * @brief A stub to mock the retrieval of current time.
26 */
27 #include "stdint.h"
28 #include "get_time_stub.h"
29
30
31 uint32_t GetCurrentTimeStub( void )
32 {
33 /* The HTTP relies on this timestamp in order to complete the network send
34 * and receive loops. Returning an unbounded timestamp does not add value to
35 * the proofs as the HTTP library uses this timestamp only in an arithmetic
36 * operation to get the difference. In C, arithmetic operations on unsigned
37 * integers are guaranteed to reliably wrap around with no adverse side
38 * effects. If the time returned was unbounded, the network send and receive
39 * loops could be unwound a large number of times making the proof execution
40 * very long. */
41 static uint32_t globalEntryTime = 0;
42
43 return globalEntryTime++;
44 }
