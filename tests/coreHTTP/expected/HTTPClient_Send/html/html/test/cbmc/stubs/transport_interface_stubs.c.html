25 int32_t TransportInterfaceSendStub( NetworkContext_t * pNetworkContext,
26 void * pBuffer,
27 size_t bytesToSend )
28 {
29 /* The number of tries to send the message before this invocation */
30 static int32_t tries = 0;
31 /* The number of bytes considered sent after this invocation */
32 int32_t ret;
33
34 __CPROVER_assert( pBuffer != NULL,
35 "TransportInterfaceSendStub pBuffer is NULL" );
36
37 /****************************************************************
38 * The send method sends some portion of the message and returns the
39 * total number of bytes in the prefix sent so far. The send method
40 * is used in a loop of the form
41 *
42 * while ( send( conn, msg, len ) < len ) { ... }
43 *
44 * We need to bound the number of loop iterations, so we need to
45 * bound the number of times it takes for send to finish sending the
46 * message. We use a static variable 'tries' to count the number of
47 * times send has tried to send the message, and we force send to
48 * finish the message after MAX_TRIES tries.
49 ****************************************************************/
50
51 if( bytesToSend <= INT32_MAX )
52 {
53 __CPROVER_assume( ret <= ( int32_t ) bytesToSend );
54 }
55
56 tries++;
57
58 if( tries >= MAX_TRIES )
59 {
60 tries = 0;
61
62 /* In order to stop the looping on send we must return an error or
63 * bytesToSend. We return an error instead of bytesToSend because
64 * bytesToSend may be a value larger than INT32_MAX. */
65 ret = -1;
66 }
67
68 return ret;
69 }
