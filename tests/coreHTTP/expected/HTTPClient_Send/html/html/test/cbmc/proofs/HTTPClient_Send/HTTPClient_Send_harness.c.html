23 /**
24 * @file HTTPClient_Send_harness.c
25 * @brief Implements the proof harness for HTTPClient_Send function.
26 */
27
28 #include "core_http_client.h"
29 #include "http_cbmc_state.h"
30 #include "transport_interface_stubs.h"
31 #include "get_time_stub.h"
32
33 void HTTPClient_Send_harness()
34 {
35 HTTPRequestHeaders_t * pRequestHeaders;
36 HTTPResponse_t * pResponse;
37 TransportInterface_t * pTransportInterface;
38 uint8_t * pRequestBodyBuf;
39 size_t reqBodyBufLen;
40 uint32_t sendFlags;
41
42 /* Initialize and make assumptions for request headers. */
43 pRequestHeaders = allocateHttpRequestHeaders( NULL );
44 __CPROVER_assume( isValidHttpRequestHeaders( pRequestHeaders ) );
45
46 /* Initialize and make assumptions for buffer to receive request body. */
47 __CPROVER_assume( reqBodyBufLen < CBMC_MAX_OBJECT_SIZE );
48 pRequestBodyBuf = mallocCanFail( reqBodyBufLen );
49
50 /* Initialize and make assumptions for response object. */
51 pResponse = allocateHttpResponse( NULL );
52 __CPROVER_assume( isValidHttpResponse( pResponse ) );
53
54 /* Initialize transport interface. */
55 pTransportInterface = allocateTransportInterface( NULL );
56
57 if( pTransportInterface != NULL )
58 {
59 /* Ideally, we want to set the function pointers below with __CPROVER_assume()
60 * but doing so makes CBMC run out of memory. */
61 pTransportInterface->send = nondet_bool() ? NULL : TransportInterfaceSendStub;
62 pTransportInterface->recv = nondet_bool() ? NULL : TransportInterfaceReceiveStub;
63 }
64
65 if( pResponse != NULL )
66 {
67 pResponse->getTime = nondet_boot() ? NULL : GetCurrentTimeStub;
68 }
69
70 HTTPClient_Send( pTransportInterface,
71 pRequestHeaders,
72 pRequestBodyBuf,
73 reqBodyBufLen,
74 pResponse,
75 sendFlags );
76 }
