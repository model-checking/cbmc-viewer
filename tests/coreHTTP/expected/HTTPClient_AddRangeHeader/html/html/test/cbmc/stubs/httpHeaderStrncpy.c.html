23 /**
24 * @file httpHeaderStrncpy.c
25 * @brief Creates a stub for httpHeaderStrncpy so that the proofs for
26 * HTTPClient_AddHeader, HTTPClient_AddRangeHeader, and
27 * HTTPClient_InitializeRequestHeaders run much faster. This stub checks if, for
28 * the input copy length, the destination and source are valid accessible
29 * memory.
30 */
31
32 #include <string.h>
33 #include <stdint.h>
34
35 void * httpHeaderStrncpy( char * pDest,
36 const char * pSrc,
37 size_t len,
38 uint8_t isField )
39 {
40 __CPROVER_assert( __CPROVER_w_ok( pDest, len ), "write" );
41 __CPROVER_assert( __CPROVER_r_ok( pSrc, len ), "read" );
42 return pDest;
43 }
