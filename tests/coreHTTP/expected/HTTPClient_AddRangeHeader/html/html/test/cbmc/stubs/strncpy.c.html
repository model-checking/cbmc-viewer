23 /**
24 * @file strncpy.c
25 * @brief Creates a stub for strncpy so that the proofs for HTTPClient_AddHeader,
26 * HTTPClient_AddRangeHeader, and HTTPClient_InitializeRequestHeaders, run much
27 * faster. This stub checks if, for the input copy length, the destination and
28 * source are valid accessible memory.
29 */
30
31 #include <string.h>
32
33 /* This is a clang macro not available on linux */
34 #ifndef __has_builtin
35 #define __has_builtin( x ) 0
36 #endif
37
38 #if __has_builtin( __builtin___strncpy_chk )
39 void * __builtin___strncpy_chk( void * dest,
40 const void * src,
41 size_t n,
42 size_t os )
43 {
44 __CPROVER_assert( __CPROVER_w_ok( dest, n ), "write" );
45 __CPROVER_assert( __CPROVER_r_ok( src, n ), "read" );
46 return dest;
47 }
48 #else
49 void * strncpy( void * dest,
50 const void * src,
51 size_t n )
52 {
53 __CPROVER_assert( __CPROVER_w_ok( dest, n ), "write" );
54 __CPROVER_assert( __CPROVER_r_ok( src, n ), "read" );
55 return dest;
56 }
57 #endif /* if __has_builtin( __builtin___strncpy_chk ) */
