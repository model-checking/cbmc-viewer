23 /**
24 * @file HTTPClient_AddRangeHeader_harness.c
25 * @brief Implements the proof harness for HTTPClient_AddRangeHeader function.
26 */
27
28 #include "core_http_client.h"
29
30 #include "http_cbmc_state.h"
31
32 void HTTPClient_AddRangeHeader_harness()
33 {
34 HTTPRequestHeaders_t * pRequestHeaders;
35 int32_t rangeStartOrlastNbytes;
36 int32_t rangeEnd;
37
38 /* Initialize and make assumptions for request headers. */
39 pRequestHeaders = allocateHttpRequestHeaders( NULL );
40 __CPROVER_assume( isValidHttpRequestHeaders( pRequestHeaders ) );
41
42 HTTPClient_AddRangeHeader( pRequestHeaders, rangeStartOrlastNbytes, rangeEnd );
43 }
