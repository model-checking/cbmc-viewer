40 /**
41 * @brief Transport interface stub for mocking data sent over the network.
42 *
43 * @param[in] pNetworkContext Implementation-defined network context.
44 * @param[in] pBuffer Buffer containing the bytes to send over the network stack.
45 * @param[in] bytesToSend Number of bytes to send over the network.
46 *
47 * @return The number of bytes sent or a negative error code.
48 */
49 int32_t TransportInterfaceSendStub( NetworkContext_t * pNetworkContext,
50 void * pBuffer,
51 size_t bytesToSend );
52
53 /**
54 * @brief Transport interface for mocking data received over the network.
55 *
56 * @param[in] pNetworkContext Implementation-defined network context.
57 * @param[in] pBuffer Buffer to receive the data into.
58 * @param[in] bytesToRecv Number of bytes requested from the network.
59 *
60 * @return The number of bytes received or a negative error code.
61 */
62 int32_t TransportInterfaceReceiveStub( NetworkContext_t * pNetworkContext,
63 void * pBuffer,
64 size_t bytesToRecv );
65
66 #endif /* ifndef TRANSPORT_INTERFACE_STUBS_H_ */
