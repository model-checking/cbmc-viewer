28 #include "http_cbmc_state.h"
29 #include "http_parser.h"
30 #include "core_http_client.h"
31
32 void httpParserOnStatusCallback_harness()
33 {
34 http_parser * pHttpParser;
35 HTTPParsingContext_t * pParsingContext;
36 HTTPResponse_t * pResponse;
37 size_t length, locOffset;
38 char * pLoc;
39
40 pHttpParser = allocateHttpSendParser( NULL );
41
42 pParsingContext = ( HTTPParsingContext_t * ) pHttpParser->data;
43
44 pResponse = pParsingContext->pResponse;
45 __CPROVER_assume( locOffset <= pResponse->bufferLen );
46 __CPROVER_assume( length <= pResponse->bufferLen - locOffset );
47 pLoc = pResponse->pBuffer + locOffset;
48
49 __CPROVER_file_local_core_http_client_c_httpParserOnStatusCallback( pHttpParser, pLoc, length );
50 }
